const DEMO_MIN_BASIC = 22;
const DEMO_MIN_PRO = 4;
const DEMO_EMAILS = ['hannah@innovatr.co.za', 'richard@innovatr.co.za'];

function ensureDemoMinimumCredits(user, credits) {

    ...credits,
