  document.body.style.overflow='hidden';
}
function closeModal(phase){
  document.getElementById('modal-'+phase).classList.add('hidden');
  document.body.style.overflow='';
}

// Close on backdrop click
document.querySelectorAll('.modal-overlay').forEach(overlay=>{
  overlay.addEventListener('click',e=>{
    if(e.target===overlay) overlay.classList.add('hidden');
  });
});

// Sub-tab switching
function switchTab(el, phase, tab){
  const modal=document.getElementById('modal-'+phase);
  modal.querySelectorAll('.modal-tab').forEach(t=>t.classList.remove('active'));
  el.classList.add('active');
  // Hide all tab panes
  modal.querySelectorAll('[id^="'+phase+'-tab-"]').forEach(p=>p.classList.add('hidden'));
  const target=document.getElementById(phase+'-tab-'+tab);
  if(target) target.classList.remove('hidden');
}

function switchExploreToSandbox(){
  const tabs=document.querySelectorAll('#modal-explore .modal-tab');
  tabs.forEach(t=>t.classList.remove('active'));
  tabs[1].classList.add('active');
  document.querySelectorAll('[id^="explore-tab-"]').forEach(p=>p.classList.add('hidden'));
  document.getElementById('explore-tab-sandbox').classList.remove('hidden');
}

// Filter button toggle
document.querySelectorAll('.filter-btn').forEach(btn=>{
  btn.addEventListener('click',function(){
    const siblings=[...this.parentElement.children].filter(c=>c.classList.contains('filter-btn'));
    siblings.forEach(s=>s.classList.remove('active'));
    this.classList.add('active');
  });
});

// Option pill toggle
document.querySelectorAll('.option-pill').forEach(pill=>{
  pill.addEventListener('click',function(){
    this.classList.toggle('selected');
  });
});

    this.closest('.study-type-cards').querySelectorAll('.stype-card').forEach(c=>c.classList.remove('selected'));

  });
});
